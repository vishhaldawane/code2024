CREATE TABLE AUTHORS
(
    AUTHORID  NUMBER(3),
    AUTHORNAME VARCHAR2(20),
    BID   NUMBER(3)
)
CREATE TABLE BOOKS
(
    BOOKID  NUMBER(3),
    BOOKNAME VARCHAR2(20),
    AID   NUMBER(3)
)

SELECT BOOKNAME,AUTHORNAME FROM  books right outer join authors

--left outer join

    on BOOKS.BOOKID = AUTHORS.BID

-- full outer join
SELECT BOOKNAME,AUTHORNAME FROM  books full outer join authors
    on BOOKS.BOOKID = AUTHORS.BID
